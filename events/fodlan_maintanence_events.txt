namespace = FOD_maintenance

FOD_maintenance.1 = { #holy orders
	type = character_event
	hidden = yes
    
	immediate = {
		every_living_character = {
			limit = {
				any_held_title = {
					this = title:d_knights_seiros
				}
			}
			save_scope_as = leader
		}
	}
	
	option = {
		hidden_effect = {
			if = {
				limit = { NOT = { title:b_knights.holder = scope:leader } }
				create_title_and_vassal_change = {
					type = leased_out
					save_scope_as = change
					add_claim_on_loss = no
				}
				title:b_knights = {
					change_title_holder_include_vassals = {
						holder = scope:leader
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
			}
		}

		create_holy_order = {
			leader = scope:leader
			capital = title:b_knights
			save_scope_as = templar_holy_order
		}
		
		hidden_effect = {
			scope:leader = {
				add_gold = 100 #So that they have some money to lend out
				add_piety_level = 2
				add_gold = holy_order_starting_gold
				every_courtier = {
					add_trait = order_member
				}
			}
		}
		scope:templar_holy_order = {
			copy_title_history = title:d_knights_seiros
		}
		hidden_effect = {
			character:100005 = { 
				change_liege = 100001
			}
			character:100011 = { 
				change_liege = 100001
			}
			character:100006 = { 
				change_liege = 100001
			}
			character:100004 = { 
				change_liege = 100001
			}
		}
	}
}

namespace = FOD_breeding

FOD_breeding.1 = { #dragons
	type = empty
	hidden = yes

	trigger = {
		OR = {
			mother = { 
				has_trait = child_goddess
			}
			real_father = { 
				has_trait = child_goddess
			}
		}
	}

	immediate = {
		if = {
			limit = {
				mother = { has_trait = child_goddess }
				real_father = { has_trait = child_goddess }
			}
			add_trait = child_goddess
		}
		else = {
			add_trait = dragon_blood
		}
	}
}
